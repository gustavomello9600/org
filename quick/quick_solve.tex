\begin{equation*}
  \begin{aligned}
  & \begin{cases}
    f(0) = 8 \\
    f(x + 1) = 4f(x) - 1 \\
  \end{cases} \\
  \\
  & \begin{cases}
    a_0 = 8 \\
    a_{n + 1} = 4a_n - 1 \\
  \end{cases} \\
  \\
  & f(1) = a_1 = 4a_0 - 1 = 4 \times 8 - 1 = 31 \\
  & f(2) = a_2 = 4a_1 - 1 = 4 \times 31 - 1 = 123 \\
\end{aligned}
\end{equation*}
